1. var

var a; var a = 10; ( Initialization and Declaration or Asignemnt)
var a = 15; ( Re-Declaration is possible)
a = 25; ( Re-Assignment is possible)
console.log(a);

2. let

let deepak_marks = 90 (Initialization and Declaration or Asignemnt)
let deepak_marks = 40 (Re-Declaration is not possible)
deepak_marks = 75 (Re-assignemnt Possible)
console.log(deepak_marks);

3. const

const fav_Person = "Arma"; ( Initialization and Declaration or Asignemnt)
const fav_Person = "Tina" ( Re-Declaration is not possible)
